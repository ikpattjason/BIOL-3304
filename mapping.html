import pandas as pd
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

df = pd.read_csv('https://raw.githubusercontent.com/ikpattjason/BIOL-3304/refs/heads/main/minus%20marcus%20practice.csv')

latitudes = df['Latitude']
longitudes = df['Longitude']

plt.figure(figsize=(10, 10))
m = Basemap(projection='merc',
            llcrnrlat=29, urcrnrlat=33,  
            llcrnrlon=-102, urcrnrlon=-93,  
            resolution='i')

m.drawcoastlines()
m.drawstates()
m.drawcountries()

x, y = m(longitudes.values, latitudes.values)

m.scatter(x, y, marker='o', color='red', zorder=5)

plt.title("Latitude and Longitude Points in Travis County, Texas")
plt.show()

import folium

latitudes = df['Latitude']
longitudes = df['Longitude']


m = folium.Map(location=[30.2672, -97.7431], zoom_start=10) 

for lat, lon in zip(latitudes, longitudes):
